python에서 hdf5 file 읽기


# h5py 설치

    $ conda install h5py
  

# hdf5 file 읽기

    >>> import h5py
    >>> f = h5py.File('filename.h5', 'r')   # 하위 group들 존재할 수 있음 - 계속 내려가야 함
    
    >>> g = f[group key]
    
    >>> list(g.keys())                      #key list 출력
    >>> list(g.items())                     #item list 출력
    >>> g[data key][:]                      #data 내용 출력
